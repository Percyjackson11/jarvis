<script>
// Check if the browser supports the required APIs  
if (navigator.mediaDevices && window.AudioContext) {  
    // Create a new AudioContext  
    const audioContext = new AudioContext();  

    // Function to start the process  
    function startDelayedPlayback() {  
        navigator.mediaDevices.getUserMedia({ audio: true, video: false })  
            .then(stream => {  
                const source = audioContext.createMediaStreamSource(stream);  
                const delay = audioContext.createDelay(1.0); // Delay in seconds  
                delay.delayTime.value = 1; // Set delay time to 3 seconds  

                // Connect the source to the delay node and the delay node to the destination  
                source.connect(delay);  
                delay.connect(audioContext.destination);  
            })  
            .catch(err => {  
                console.error('Error accessing the microphone', err);  
            });  
    }  

    // Start delayed playback  
    startDelayedPlayback();  
} else {  
    console.log('Your browser does not support the required APIs.');  
}
</script>
